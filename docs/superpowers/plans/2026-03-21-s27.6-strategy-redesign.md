```python
class DualMomentumStrategy(BaseStrategy):
    """
    Dual Momentum (Antonacci) — absolute + relative momentum.

    - Absolute: Is the asset's N-month return positive? If not → HOLD (cash)
    - Relative: Of assets with positive absolute momentum, which is strongest?
    - Lookback: configurable, default 252 daily candles (~12 months)
    - Trades ~12 times/year at daily timeframe
    """

    @property
    def name(self) -> str:
        return f"Dual Momentum ({self.lookback_period}d)"

    @property
    def strategy_type(self) -> str:
        return "dual_momentum"

    def analyze(self, df, portfolio_value, timestamp=None) -> Signal:
        if len(df) < self.lookback_period:
            return Signal(signal_type=SignalType.HOLD)

        # Absolute momentum: is N-period return positive?
        current_price = float(df.iloc[-1]["close"])
        lookback_price = float(df.iloc[-self.lookback_period]["close"])
        absolute_return = (current_price - lookback_price) / lookback_price

        if absolute_return <= 0:
            # Negative absolute momentum → go to cash
            return Signal(signal_type=SignalType.SELL, confidence=0.7,
                         reason=f"Negative absolute momentum: {absolute_return:.1%} over {self.lookback_period} periods")

        # Positive absolute momentum → BUY with confidence based on strength
        confidence = min(1.0, 0.5 + abs(absolute_return) * 2)
        return Signal(signal_type=SignalType.BUY, confidence=confidence,
                     quantity=portfolio_value * 0.15 / current_price,
                     reason=f"Positive momentum: {absolute_return:.1%} over {self.lookback_period} periods")
```
